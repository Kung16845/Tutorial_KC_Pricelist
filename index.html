Fn 1 calc_multi_discount(base_price, discount_str)

คำนวณส่วนลดแบบ multi-tier — คืน tuple (discount_amount, net_price)

✓ รองรับ 2 รูปแบบ: '30%' = ลด % | '200' = ลบ flat 200 บาท | '30%+200' = ผสม
อธิบายทุกบรรทัดPython
def calc_multi_discount(base_price, discount_str):

    if not discount_str or discount_str in ('-', '0', '0.0', ''):
        return 0.0, base_price    ← ไม่มีส่วนลด → return ทันที

    try:
        remaining = base_price         ← เริ่มจากราคาเต็ม แล้วหักทีละขั้น
        for p in discount_str.replace(' ', '').split('+'):
                 ↑ ลบ space → split ด้วย '+' → ['30%', '200', '5%']
            if not p: continue
            if '%' in p:
                remaining *= (1.0 - float(p.replace('%', '')) / 100.0)
                            ↑ '30%' → remaining × 0.70
            else:
                remaining -= float(p)
                           ↑ '200' → remaining − 200 บาท
        return base_price - remaining, remaining
        ↑ return (จำนวนที่ลด, ราคาหลังลด)
    except Exception:
        return 0.0, base_price  ← parse ผิดพลาด → safe fallback ไม่ลดเลย
discount_strbase_pricereturn
''1,000(0.0, 1000.0)
'30%'1,000(300.0, 700.0)
'50%+20%'1,000(600.0, 400.0) ← ×0.5 × 0.8
'30%+200'1,000(500.0, 500.0) ← ×0.7 แล้ว −200
'150'1,000(150.0, 850.0)
Fn 2 get_component_qty(c_desc, default_qty=1.0, prefix_match=False)

ดึง จำนวน (qty) ของ Component จาก BOM Preview
ค้นแบบ exact match จาก description / name / default_code (default)
หรือแบบ prefix match จาก default_code เมื่อส่ง prefix_match=True

อธิบายทุกบรรทัดPython
def get_component_qty(c_desc, default_qty=1.0, prefix_match=False):

    for c in components:           ← วน list BOM Preview (qty ที่ BOM คำนวณแล้ว)
        if prefix_match:            ← โหมด prefix — ค้นจาก default_code เท่านั้น
            if (c.get('default_code') or '').startswith(c_desc):
                return float(c.get('qty', default_qty))
        elif c_desc in (            ← โหมดปกติ — exact match ทุก field
            c.get('description') or '',
            c.get('name')        or '',
            c.get('default_code') or ''
        ):
            return float(c.get('qty', 0.0))

    return default_qty   ← ไม่เจอ → ใช้ default

def get_component_sales_price(c_desc, default_price=0.0):

    for c in components:
        if c_desc in (c.get('description') or '',
                      c.get('name')        or '',
                      c.get('default_code') or ''):
            prod_id = c.get('product_id')
            if prod_id:
                prod = env['product.product'].browse(prod_id)
                if current_pricelist:
                    try:
                        return current_pricelist.get_product_price(prod, 1.0, current_partner)
                                                   ↑ ราคาตาม pricelist (qty=1)
                    except Exception:
                        return prod.list_price or default_price
                return prod.list_price or default_price
    return default_price
Fn 4 get_price_from_pricing_line(pl)

def get_price_from_pricing_line(pl):

    if not pl:
        return 0.0, 0.0, ''   ← ป้องกัน None record

    if   pricelist_name == 'credit':
        return pl.credit,       pl.addons_price_credit,       pl.discount_credit       or ''
    elif pricelist_name == 'cash':
        return pl.cash,         pl.addons_price_cash,         pl.discount_cash         or ''
    elif pricelist_name == 'homeowner':
        return pl.homeowner,    pl.addons_price_homeowner,    pl.discount_homeowner    or ''
    elif pricelist_name == 'modern_trade':
        return pl.modern_trade, pl.addons_price_modern_trade, pl.discount_modern_trade or ''
    else:                    ← default: online
        return pl.online,      pl.addons_price_online,       pl.discount_online       or ''

Fn 5 get_attr_value_float(attr_name=None, attr_id=None, default=0.0)

ดึงค่าตัวเลขจาก Attribute — รองรับ 2 วิธีค้นหา (ชื่อ / id) และ 2 แหล่งข้อมูล (ค่าที่เลือก / custom_value)

อธิบายทุกบรรทัดPython
def get_attr_value_float(attr_name=None, attr_id=None, default=0.0):
    raw = ''

    # วิธีที่ 1: ค้นจากชื่อ Attribute
    if attr_name:
        raw = values.get(attr_name, '')
        if raw in ('ระบุ', 'custom', '') and wizard:
            ↑ 'ระบุ' = ผู้ใช้พิมพ์เอง → ต้องอ่าน custom_value จาก wizard
            try:
                _found = [_l for _l in wizard.attr_line_ids
                          if _l.attribute_id.name == attr_name]
                if _found and _found[0].custom_value:
                    raw = _found[0].custom_value
            except Exception: pass

    # วิธีที่ 2: ค้นจาก attr_id (reliable กว่า ถ้าชื่ออาจเปลี่ยน)
    if not raw and attr_id is not None and wizard:
        try:
            _found = [_l for _l in wizard.attr_line_ids
                      if _l.attribute_id.id == attr_id]   ← match by ID
            if _found:
                raw = _found[0].custom_value or (
                      _found[0].value_id.name if _found[0].value_id else '')
        except Exception: pass

    if raw:
        try:
            return float(str(raw).replace(',', '').strip())
                               ↑ '2,500' → '2500' → 2500.0
        except (ValueError, TypeError):
            return default
    return default

    elif pricelist_name == 'homeowner':    return pl.homeowner,    pl.addons_price_homeowner,    pl.discount_homeowner    or ''
    elif pricelist_name == 'modern_trade': return pl.modern_trade, pl.addons_price_modern_trade, pl.discount_modern_trade or ''
    else:                                   return pl.online,       pl.addons_price_online,       pl.discount_online       or ''
